%%% Depth of Field

\begin{frame}[plain]{}
  \vspace{3ex}
  \begin{center} \LARGE \bf
    Depth of Field
  \end{center}

  {\Large 
    $$\mathit{d = d_1 + d_2 = \frac{2NcD^2f^2}{f^4 - N^2c^2D^2}}$$
  }

  \vspace{2ex}
  {\scriptsize
    $$\mathit{d_1 = \frac{NcD^2}{f^2 + NCD}}, \, \, \, \mathit{d_2 = \frac{NcD^2}{f^2 - NCD}}$$
  }

\end{frame}
